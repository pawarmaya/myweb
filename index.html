Creating a Jenkins Pipeline for Docker Swarm

Step 1: Create a New Jenkins Pipeline

pipeline { agent any }

Step 2: Define Stages

stages {
    stage('Build Docker Image') {
        steps {
            echo 'Building Docker image...'
        }
    }
}

Step 3: Add Credentials Management

stage('Docker Login') {
    steps {
        withCredentials([usernamePassword(credentialsId: 'docker-hub-credentials', usernameVariable: 'DOCKER_USERNAME', passwordVariable: 'DOCKER_PASSWORD')]) {
            sh 'echo $DOCKER_PASSWORD | docker login -u $DOCKER_USERNAME --password-stdin'
        }
    }
}

Step 4: Add Push and Deployment Steps

stage('Push Docker Image') {
    steps {
        sh 'docker push maya2711/myweb'
    }
}

Step 5: Add Post Actions

post {
    success {
        echo 'Pipeline completed successfully!'
    }
}
